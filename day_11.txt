// Question 1 - 3 sum problem leetcode 
#include<iostream>
using namespace std;
int arr_sum(int arr[], int n, int s){

   for(int i=0;i<n; i++ ){
       
   		for(int j = i+1; j<n; j++) {

                  for (int k = j + 1; k <n; k++) {

                        if (arr[i] + arr[j] + arr[k] == s) {
                            cout << arr[i] << " " << arr[j] << " " << arr[k] << endl;

                    }
                  }
                }
   }
}
